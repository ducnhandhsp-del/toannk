// --- TIỆN ÍCH GỘP CLASS TAILWIND ---
function cn(...inputs: ClassValue[]) {
  return twMerge(clsx(inputs));
}
// Thêm hàm này vào đầu App.tsx (sau formatDisplayDate)
function parseScheduleSlots(cls: any): { thu: string; gio: string }[] {
  const results: { thu: string; gio: string }[] = [];
  const dayMap: Record<string, string> = {
    'T2': 'Thứ 2', 'T3': 'Thứ 3', 'T4': 'Thứ 4',
    'T5': 'Thứ 5', 'T6': 'Thứ 6', 'T7': 'Thứ 7', 'CN': 'CN'
  };
  ['Buổi 1', 'Buổi 2', 'Buổi 3'].forEach(key => {
    const val = (cls[key] || '').trim();
    if (!val || val === '—') return;
    // Format: "T4 9:00–10:30"
    const match = val.match(/^(T\d|CN)\s+(.+)$/);
    if (match) {
      results.push({ thu: dayMap[match[1]] || match[1], gio: match[2] });
    }
  });
  return results;
}
// --- TIỆN ÍCH: Chuyển ISO (2026-03-05) → DD/MM/YYYY để lưu vào sheet ---
function formatDisplayDate(isoDate: string): string {
  if (!isoDate) return '';
  const parts = isoDate.split('-');
  if (parts.length === 3) return `${parts[2]}/${parts[1]}/${parts[0]}`;
  return isoDate;
}
