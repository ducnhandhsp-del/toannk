import React, { useState, useEffect, useMemo } from 'react';
import { 
  GraduationCap, LayoutDashboard, Users, BookOpen, UserCheck, 
  CreditCard, FileText, Save, Printer, X, Check, 
  Plus, Calendar, TrendingUp, PieChart as PieChartIcon, 
  History, BarChart2, LayoutGrid, List, PhoneCall, Search, 
  Filter, BookText, DollarSign, Send
} from 'lucide-react';
import { 
  BarChart, Bar, XAxis, YAxis, CartesianGrid, Tooltip, 
  ResponsiveContainer, PieChart, Pie, Cell, Legend, LabelList
} from 'recharts';
import { clsx, type ClassValue } from 'clsx';
import { twMerge } from 'tailwind-merge';

// --- TIỆN ÍCH GỘP CLASS TAILWIND ---
function cn(...inputs: ClassValue[]) {
  return twMerge(clsx(inputs));
}

// --- CẤU HÌNH API GOOGLE SCRIPTS ---
const SCRIPT_URL = 'https://script.google.com/macros/s/AKfycbwDDCm7fkcHehvZxMSlqQVlARcJWtQu_qRu33SPWn4XEICAJuOxREqkfo-5kQW-4Flp6w/exec';

// --- ĐỊNH NGHĨA KIỂU DỮ LIỆU (INTERFACE) ---
// BỔ SUNG THÊM 2 TAB: 'diary' (Nhật ký) và 'finance' (Tài chính)
type Screen = 'overview' | 'teaching' | 'diary' | 'students' | 'finance' | 'fees';

interface Student {
  id: string; name: string; dob: string; branch: string; grade: string; school: string; teacher: string; 
  parentName: string; parentPhone: string; studentPhone: string; address: string; academicLevel: string; 
  goal: string; supportNeeded: string; classId: string; startDate: string; endDate: string;
}

interface S1aRecord {
  id: string; date: string; docNum: string; description: string; studentId: string; 
  studentName: string; payer: string; method: string; amount: number; note?: string;
}

const COLORS = ['#14b8a6', '#3b82f6', '#f59e0b', '#ef4444', '#8b5cf6'];

export default function App() {
  // --- STATES QUẢN LÝ DỮ LIỆU TỔNG ---
  const [activeScreen, setActiveScreen] = useState<Screen>('overview');
  const [loading, setLoading] = useState(true);
  const [isSaving, setIsSaving] = useState(false);
  
  const [students, setStudents] = useState<Student[]>([]);
  const [classes, setClasses] = useState<any[]>([]);
  const [payments, setPayments] = useState<S1aRecord[]>([]);
  const [teachingLogs, setTeachingLogs] = useState<any[]>([]);
  const [attendanceLogs, setAttendanceLogs] = useState<any[]>([]); // Dữ liệu điểm danh gốc
  const [financeSummary, setFinanceSummary] = useState<any>(null);

  // --- STATES BỘ LỌC & GIAO DIỆN ---
  const [searchTerm, setSearchTerm] = useState('');
  const [filterClass, setFilterClass] = useState('');
  const [studentViewMode, setStudentViewMode] = useState<'table' | 'grid'>('table');
  
  // MỚI: State cho Tab Tài chính
  const [financeFilterMonth, setFinanceFilterMonth] = useState(`${(new Date().getMonth() + 1).toString().padStart(2, '0')}/${new Date().getFullYear()}`);
  const [financeFilterTeacher, setFinanceFilterTeacher] = useState('');

  // --- STATES GIAO DIỆN MODALS ---
  const [selectedClassId, setSelectedClassId] = useState('');
  const [showInvoice, setShowInvoice] = useState<S1aRecord | null>(null);
  const [showStudentDetail, setShowStudentDetail] = useState<Student | null>(null);
  const [showAddHSModal, setShowAddHSModal] = useState(false);
  const [showAddFeeModal, setShowAddFeeModal] = useState(false);
  const [editingStudent, setEditingStudent] = useState<Student | null>(null);

  // --- STATES GIẢNG DẠY ---
  const [teachingDate, setTeachingDate] = useState(new Date().toISOString().split('T')[0]);
  const [attendance, setAttendance] = useState<Record<string, string>>({});
  const [attendanceNotes, setAttendanceNotes] = useState<Record<string, string>>({});
  const [journalContent, setJournalContent] = useState('');
  const [journalHomework, setJournalHomework] = useState('');

  // --- STATES NHẬP LIỆU BIỂU MẪU ---
  const [formHS, setFormHS] = useState<any>({ branch: 'Đào Tấn', academicLevel: 'Khá' });
  const [formFee, setFormFee] = useState<any>({ method: 'Chuyển khoản', date: new Date().toISOString().split('T')[0] });

  // --- HÀM ĐỊNH DẠNG NGÀY CHUẨN (DD/MM/YYYY) ---
  const formatDisplayDate = (dateStr: any) => {
    if (!dateStr) return '---';
    try {
      const d = new Date(dateStr);
      if (isNaN(d.getTime())) return dateStr;
      return `${d.getDate().toString().padStart(2, '0')}/${(d.getMonth() + 1).toString().padStart(2, '0')}/${d.getFullYear()}`;
    } catch (e) { return dateStr; }
  };
// 1. TẢI DỮ LIỆU & TÍNH TOÁN NÂNG CAO (VERSION 9.0 ULTIMATE)
  const loadData = async () => {
    try {
      setLoading(true);
      const res = await fetch(SCRIPT_URL);
      const data = await res.json();
      
      // Ánh xạ Học sinh
      const mappedHS = (data.students || []).map((s: any) => ({
        id: s["Mã HS"] || "", name: s["Họ và tên học sinh"] || "---", 
        dob: formatDisplayDate(s["Ngày tháng năm sinh"]), branch: s["Cơ sở học tập"] || "---", 
        grade: s["Khối lớp hiện tại"] || "---", school: s["Trường đang học"] || "---",
        teacher: s["Giáo viên trực tiếp giảng dạy"] || "---", parentName: s["Họ và tên phụ huynh"] || "---",
        parentPhone: s["Số điện thoại phụ huynh (Zalo)"] || "---", studentPhone: s["Số điện thoại học sinh"] || "---",
        address: s["Địa chỉ thường trú"] || "---", academicLevel: s["Học lực môn Toán hiện tại"] || "---",
        goal: s["Mục tiêu điểm số học kỳ tới"] || "---", supportNeeded: s["Kiến thức em cần hỗ trợ thêm"] || "---",
        classId: s["Mã Lớp"] || "---", startDate: formatDisplayDate(s["Ngày bắt đầu"]), endDate: formatDisplayDate(s["Ngày kết thúc"])
      }));
      setStudents(mappedHS);
      setClasses(data.classes || []);
      
      // Lưu trữ dữ liệu điểm danh gốc để lát nữa hiển thị vào Hồ sơ 360 độ
      setAttendanceLogs(data.attendanceLogs || []);

      // Ánh xạ Học phí
      const mappedPayments = (data.payments || []).map((p:any, i:number) => {
        const foundStudent = mappedHS.find((s:any) => s.id === p["Mã HS"]);
        return {
          id: i.toString(), date: formatDisplayDate(p["Ngày CT"]), docNum: p["Số hiệu CT"] || "---",
          studentId: p["Mã HS"] || "---", studentName: foundStudent ? foundStudent.name : "HS KHÔNG TỒN TẠI",
          payer: p["Người thanh toán"] || "---", method: p["Hình thức"] || "---", 
          description: p["Diễn giải"] || "---", amount: Number(p["Số tiền"]) || 0
        };
      });
      setPayments(mappedPayments);

      // TÍNH TOÁN DASHBOARD: Bảng Lớp học (Đếm sĩ số & Tìm Giáo viên)
      const classStats = (data.classes || []).map((c:any) => {
        const cId = c["Mã Lớp"];
        const stds = mappedHS.filter((s:any) => s.classId === cId);
        const teacher = stds.length > 0 ? stds[0].teacher : "---";
        return { id: cId, studentCount: stds.length, teacher };
      });

      // TÍNH TOÁN DASHBOARD: Biểu đồ cột đôi Doanh thu theo tháng (Định dạng X-Axis chuẩn báo cáo: M/YY)
      const monthlyRevMap: Record<string, { Nhan: number, Kien: number }> = {};
      mappedPayments.forEach((p:any) => {
        if (p.date && p.date !== '---') {
          // p.date có dạng DD/MM/YYYY. Ta tách ra để lấy M/YY (VD: 3/26)
          const parts = p.date.split('/');
          if(parts.length === 3) {
            const monthStr = `${parseInt(parts[1], 10)}/${parts[2].substring(2)}`; 
            if (!monthlyRevMap[monthStr]) monthlyRevMap[monthStr] = { Nhan: 0, Kien: 0 };
            
            const student = mappedHS.find((s:any) => s.id === p.studentId);
            const teacher = student ? student.teacher : "";
            if (teacher.includes("Nhân")) monthlyRevMap[monthStr].Nhan += p.amount;
            else if (teacher.includes("Kiên")) monthlyRevMap[monthStr].Kien += p.amount;
          }
        }
      });
      const monthlyRevenueChart = Object.keys(monthlyRevMap).map(key => ({
        month: key, Nhan: monthlyRevMap[key].Nhan, Kien: monthlyRevMap[key].Kien
      })).reverse(); // Đảo ngược để hiện từ quá khứ đến hiện tại

      // XỬ LÝ NHẬT KÝ DẠY HỌC: Kết nối Điểm danh (dựa vào Mã Lớp) & Bài dạy
      const processedDiary = (data.teachingLogs || []).map((log:any) => {
        const logDateFormatted = formatDisplayDate(log["Ngày"]);
        const classId = log["Mã Lớp"];
        
        // Lọc các bản ghi điểm danh khớp cả Ngày và Mã Lớp
        const logAtt = (data.attendanceLogs || []).filter((a:any) => formatDisplayDate(a["Ngày"]) === logDateFormatted && a["Mã Lớp"] === classId);
        
        return {
          date: logDateFormatted, classId: classId,
          content: log["Nội dung bài dạy"] || "---", homework: log["Bài tập về nhà"] || "---",
          present: logAtt.filter((a:any) => a["Trạng thái"] === "Có mặt").length,
          absent: logAtt.filter((a:any) => a["Trạng thái"] === "Vắng").length,
          late: logAtt.filter((a:any) => a["Trạng thái"] === "Muộn").length,
          teacherName: log["Giáo viên"] || "---" // Hiển thị tên GV trong nhật ký
        };
      });
      setTeachingLogs(processedDiary);

      setFinanceSummary({ ...(data.finance || {}), classStats, monthlyRevenueChart });
      if (data.classes?.length > 0 && !selectedClassId) setSelectedClassId(data.classes[0]["Mã Lớp"]);
    } catch (e) { console.error("Lỗi:", e); } finally { setLoading(false); }
  };

  useEffect(() => { loadData(); }, []);

  // 2. LƯU GIẢNG DẠY (Gửi kèm Tên Giáo viên & Mã Lớp mới)
  const handleSaveTeaching = async () => {
    if (isSaving) return;
    const attRecords = students.filter(s => s.classId === selectedClassId).map(s => ({
      maHS: s.id, tenHS: s.name, trangThai: attendance[s.id] || 'Có mặt', ghiChu: attendanceNotes[s.id] || ''
    }));
    
    // Tự động tìm tên giáo viên phụ trách lớp này
    const currentClassStudents = students.filter(s => s.classId === selectedClassId);
    const teacherName = currentClassStudents.length > 0 ? currentClassStudents[0].teacher : "Chưa cập nhật";

    setIsSaving(true);
    try {
      const payload = { 
        action: "saveDaily", date: teachingDate, maLop: selectedClassId, teacherName: teacherName,
        attendance: attRecords, content: journalContent || "---", homework: journalHomework || "---" 
      };
      await fetch(SCRIPT_URL, { method: 'POST', body: JSON.stringify(payload) });
      setJournalContent(''); setJournalHomework(''); setAttendance({}); setAttendanceNotes({});
      alert("Đã lưu dữ liệu dạy học thành công!"); loadData();
    } catch (e) { alert("Lỗi khi lưu dữ liệu!"); } finally { setIsSaving(false); }
  };

  // 3. LƯU HỌC PHÍ
  const handleSaveFee = async () => {
    if (isSaving) return;
    setIsSaving(true);
    try {
      const now = new Date();
      const saveTime = `${now.getHours().toString().padStart(2, '0')}:${now.getMinutes().toString().padStart(2, '0')} - ${formatDisplayDate(formFee.date)}`;
      const docNum = `PT-${now.getFullYear().toString().slice(2)}${(now.getMonth() + 1).toString().padStart(2, '0')}${now.getDate().toString().padStart(2, '0')}-${formFee.maHS}`;

      await fetch(SCRIPT_URL, { method: 'POST', body: JSON.stringify({ 
        action: "savePayment", timeStamp: saveTime, soCT: docNum, ...formFee 
      })});
      setShowAddFeeModal(false); loadData();
    } catch (e) { alert("Lỗi khi lưu thu phí!"); } finally { setIsSaving(false); }
  };

  // 4. LƯU/SỬA HỌC SINH
  const handleSaveHS = async () => {
    if (isSaving) return;
    setIsSaving(true);
    try {
      await fetch(SCRIPT_URL, { method: 'POST', body: JSON.stringify({ action: editingStudent ? "updateHS" : "saveHS", ...formHS }) });
      setShowAddHSModal(false); setEditingStudent(null); loadData();
    } catch (e) { alert("Lỗi khi lưu hồ sơ!"); } finally { setIsSaving(false); }
  };
// --- GIAO DIỆN CHỜ ---
  if (loading && !isSaving) return (
    <div className="min-h-screen flex flex-col items-center justify-center bg-teal-50 text-teal-600">
      <GraduationCap size={80} className="animate-bounce mb-4" />
      <h1 className="text-xl font-black uppercase italic tracking-widest">Hệ Thống Đang Xử Lý...</h1>
    </div>
  );

  // --- COMPONENT: THANH ĐIỀU HƯỚNG (NAVBAR 6 TAB) ---
  const Navbar = () => (
    <nav className="bg-white border-b border-gray-200 sticky top-0 z-50 shadow-sm">
      <div className="max-w-7xl mx-auto px-4 flex justify-between h-20 items-center overflow-x-auto scrollbar-hide">
        <div className="flex items-center text-teal-600 shrink-0 mr-6">
          <GraduationCap size={40} className="mr-3" />
          <div className="flex flex-col">
            <span className="font-black text-xl uppercase leading-none">Lớp Toán NK</span>
            <span className="text-[10px] font-bold text-gray-400 uppercase tracking-[0.2em] mt-1">Version 9.0 Ultimate</span>
          </div>
        </div>
        <div className="flex space-x-1 shrink-0">
          {[
            { id: 'overview', label: 'Tổng quan', icon: LayoutDashboard, color: 'text-teal-600' },
            { id: 'teaching', label: 'Giảng dạy', icon: UserCheck, color: 'text-green-600' },
            { id: 'diary', label: 'Nhật ký', icon: BookText, color: 'text-indigo-600' },
            { id: 'students', label: 'Học sinh', icon: Users, color: 'text-blue-600' },
            { id: 'finance', label: 'Tài chính', icon: DollarSign, color: 'text-emerald-600' }, // TAB MỚI
            { id: 'fees', label: 'Sổ thu phí', icon: CreditCard, color: 'text-orange-500' },
          ].map((item) => (
            <button 
              key={item.id} 
              onClick={() => setActiveScreen(item.id as Screen)}
              className={cn("flex flex-col items-center px-4 py-2 rounded-2xl transition-all", activeScreen === item.id ? "bg-gray-100 " + item.color : "text-gray-400 hover:bg-gray-50")}
            >
              <item.icon size={22} /><span className="text-[10px] font-black uppercase mt-1 tracking-wider">{item.label}</span>
            </button>
          ))}
        </div>
      </div>
    </nav>
  );

  // Lọc Học sinh cho Tab Học Sinh
  const filteredStudents = students.filter(s => {
    const matchSearch = s.name.toLowerCase().includes(searchTerm.toLowerCase()) || s.id.toLowerCase().includes(searchTerm.toLowerCase());
    const matchClass = filterClass === '' || s.classId === filterClass;
    return matchSearch && matchClass;
  });

  return (
    <div className="min-h-screen bg-[#F8FAFC] pb-12 font-sans selection:bg-teal-100">
      <Navbar />
      <main className="max-w-7xl mx-auto px-4 py-8">

        {/* --- TAB 1: TỔNG QUAN (DASHBOARD CHUYÊN NGHIỆP) --- */}
        {activeScreen === 'overview' && (
          <div className="space-y-6 animate-in fade-in duration-700">
            {/* 3 THẺ SỐ LIỆU */}
            <div className="grid grid-cols-1 md:grid-cols-3 gap-6">
              <div className="bg-white p-6 rounded-[32px] shadow-sm border border-gray-100 flex items-center space-x-5">
                <div className="p-4 bg-blue-100 text-blue-600 rounded-3xl"><Users size={32}/></div>
                <div><p className="text-sm font-bold text-gray-400 uppercase">Tổng Học sinh</p><p className="text-4xl font-black text-gray-900">{students.length}</p></div>
              </div>
              <div className="bg-white p-6 rounded-[32px] shadow-sm border border-gray-100 flex items-center space-x-5">
                <div className="p-4 bg-teal-100 text-teal-600 rounded-3xl"><TrendingUp size={32}/></div>
                <div><p className="text-sm font-bold text-gray-400 uppercase">Tổng Doanh thu</p><p className="text-4xl font-black text-gray-900">{financeSummary?.totalRevenue?.toLocaleString() || 0}đ</p></div>
              </div>
              <div className="bg-white p-6 rounded-[32px] shadow-sm border border-gray-100 flex items-center space-x-5">
                <div className="p-4 bg-purple-100 text-purple-600 rounded-3xl"><BookOpen size={32}/></div>
                <div><p className="text-sm font-bold text-gray-400 uppercase">Lớp đang mở</p><p className="text-4xl font-black text-gray-900">{classes.length}</p></div>
              </div>
            </div>

            <div className="grid grid-cols-1 lg:grid-cols-2 gap-6">
              {/* BIỂU ĐỒ CỘT ĐÔI BÁO CÁO TÀI CHÍNH (CÓ SỐ TRÊN ĐỈNH) */}
              <div className="bg-white p-8 rounded-[40px] shadow-sm border border-gray-100 lg:col-span-2">
                <h3 className="font-black uppercase tracking-tight text-gray-800 mb-6 flex items-center"><BarChart2 size={20} className="mr-2 text-teal-600"/> Dòng tiền theo tháng</h3>
                <div className="h-[280px]">
                  <ResponsiveContainer>
                    <BarChart data={financeSummary?.monthlyRevenueChart || []} margin={{top: 30, right: 10, left: 10, bottom: 5}}>
                      <CartesianGrid strokeDasharray="3 3" vertical={false} horizontal={false}/>
                      <XAxis dataKey="month" tick={{fontSize: 12, fontWeight: 'bold'}} axisLine={true} tickLine={false}/>
                      <YAxis tickFormatter={(value) => `${value / 1000000}M`} width={50} tick={{fontSize: 12}} axisLine={false} tickLine={false}/>
                      <Tooltip cursor={{fill:'#f0fdfa'}} formatter={(value: number) => `${value.toLocaleString()}đ`}/>
                      <Legend verticalAlign="bottom" height={36} iconType="circle"/>
                      
                      <Bar dataKey="Nhan" name="Thầy Nhân" fill="#14b8a6" radius={[6, 6, 0, 0]} maxBarSize={60}>
                        <LabelList dataKey="Nhan" position="top" formatter={(val: number) => val > 0 ? `${val/1000000}M` : ''} style={{ fontSize: '11px', fontWeight: '900', fill: '#0d9488' }} />
                      </Bar>
                      
                      <Bar dataKey="Kien" name="Cô Kiên" fill="#f59e0b" radius={[6, 6, 0, 0]} maxBarSize={60}>
                        <LabelList dataKey="Kien" position="top" formatter={(val: number) => val > 0 ? `${val/1000000}M` : ''} style={{ fontSize: '11px', fontWeight: '900', fill: '#d97706' }} />
                      </Bar>
                    </BarChart>
                  </ResponsiveContainer>
                </div>
              </div>

              {/* BẢNG LỚP ĐANG MỞ (HIỂN THỊ SĨ SỐ & GIÁO VIÊN) */}
              <div className="bg-white rounded-[40px] border border-blue-100 overflow-hidden shadow-sm h-full">
                <div className="bg-blue-50 p-5 border-b border-blue-100"><h3 className="font-black uppercase text-blue-700 flex items-center"><BookOpen size={18} className="mr-2"/> Danh sách Lớp học</h3></div>
                <table className="w-full text-left text-sm"><tbody className="divide-y divide-gray-50">
                  {financeSummary?.classStats?.map((c:any, i:number) => (
                    <tr key={i} className="hover:bg-blue-50/20">
                      <td className="p-4 pl-6 font-black uppercase text-blue-700">{c.id}</td>
                      <td className="p-4 font-bold text-gray-600">{c.studentCount} HS</td>
                      <td className="p-4 pr-6 text-right font-black text-gray-900">{c.teacher}</td>
                    </tr>
                  ))}
                </tbody></table>
              </div>

              {/* BẢNG THU PHÍ GẦN ĐÂY */}
              <div className="bg-white rounded-[40px] border border-orange-100 overflow-hidden shadow-sm h-full">
                <div className="bg-orange-50 p-5 border-b border-orange-100"><h3 className="font-black uppercase text-orange-700 flex items-center"><CreditCard size={18} className="mr-2"/> Thu phí gần đây</h3></div>
                <table className="w-full text-left text-sm"><tbody className="divide-y divide-gray-50">
                  {payments.slice(0, 5).map((p, i) => (
                    <tr key={i} className="hover:bg-orange-50/20">
                      <td className="p-4 pl-6 font-bold text-gray-500 whitespace-nowrap">{p.date}</td>
                      <td className="p-4 font-black uppercase text-gray-900 truncate max-w-[120px]">{p.studentName}</td>
                      <td className="p-4 pr-6 text-right font-black text-orange-600">+{p.amount?.toLocaleString()}đ</td>
                    </tr>
                  ))}
                </tbody></table>
              </div>
            </div>
          </div>
        )}
{/* --- TAB 2: GIẢNG DẠY (ĐIỂM DANH & GHI SỔ) --- */}
        {activeScreen === 'teaching' && (
          <div className="space-y-6 animate-in slide-in-from-bottom-8">
            <div className="bg-green-600 p-8 rounded-[40px] text-white shadow-xl flex flex-col md:flex-row justify-between items-center gap-4 border-b-8 border-green-700">
              <div><h2 className="text-3xl font-black uppercase italic tracking-tighter">Sổ giảng dạy trực tuyến</h2><p className="text-green-100 text-base font-bold italic">Mặc định: Có mặt</p></div>
              <div className="flex items-center space-x-3 w-full md:w-auto">
                <div className="bg-white/20 px-5 py-3 rounded-2xl flex items-center shadow-inner"><Calendar size={20} className="mr-2" /><input type="date" value={teachingDate} onChange={e=>setTeachingDate(e.target.value)} className="bg-transparent text-white outline-none font-black text-lg [color-scheme:dark]" /></div>
                <select value={selectedClassId} onChange={e=>setSelectedClassId(e.target.value)} className="bg-white text-green-700 px-6 py-4 rounded-2xl font-black text-lg outline-none shadow-2xl">
                  {classes.map(c => <option key={c["Mã Lớp"]} value={c["Mã Lớp"]}>Mã Lớp: {c["Mã Lớp"] || "---"}</option>)}
                </select>
              </div>
            </div>

            <div className="grid grid-cols-1 lg:grid-cols-3 gap-6">

                <div><h2 className="text-xl font-black uppercase italic tracking-tighter">Sổ thu học phí S1a</h2><p className="text-orange-100 text-sm font-bold italic">Ghi chép chính xác dòng tiền</p></div>
                
                <div className="flex flex-col md:flex-row items-center space-y-4 md:space-y-0 md:space-x-4 w-full md:w-auto">
                  <div className="relative w-full md:w-64">
                    <div className="absolute inset-y-0 left-0 pl-4 flex items-center pointer-events-none"><Search size={18} className="text-orange-300"/></div>
                    <input type="text" placeholder="Tìm tên, mã, người nộp..." className="w-full pl-11 pr-4 py-3 bg-black/20 text-white placeholder-orange-200 rounded-xl outline-none focus:ring-2 ring-white font-bold text-sm" value={searchTerm} onChange={(e) => setSearchTerm(e.target.value)} />
                  </div>
                  <button onClick={() => setShowAddFeeModal(true)} className="w-full md:w-auto bg-white text-orange-600 px-6 py-3 rounded-xl font-black text-sm uppercase shadow-lg hover:bg-orange-50 transition-all whitespace-nowrap">+ Ghi thu mới</button>
                </div>
              </div>
              <div className="bg-white rounded-[40px] border border-orange-100 overflow-hidden shadow-xl">
                <div className="overflow-x-auto">
                  <table className="w-full text-left text-sm">
                    <thead className="bg-orange-50 text-orange-700 font-black uppercase tracking-widest border-b-2 border-orange-100">
                      <tr className="text-sm">
                        <th className="p-5 whitespace-nowrap">NGÀY CT</th><th className="p-5">MÃ CT</th><th className="p-5">NGƯỜI NỘP</th><th className="p-5">HỌC SINH</th><th className="p-5">DIỄN GIẢI</th><th className="p-5 text-right">SỐ TIỀN</th><th className="p-5 text-center">BIÊN LAI</th>
                      </tr>
                    </thead>
                    <tbody className="divide-y divide-gray-100">
                      {filteredPayments.length === 0 ? (
                         <tr><td colSpan={7} className="p-8 text-center text-gray-400 font-bold italic">Không tìm thấy giao dịch nào...</td></tr>
                      ) : (
                        filteredPayments.map((r, i) => (
                          <tr key={i} className="hover:bg-orange-50/20 transition-all">
                            <td className="p-5 font-bold text-gray-500 whitespace-nowrap">{r.date || "---"}</td><td className="p-5 font-black text-orange-500 italic text-xs">{r.docNum || "---"}</td>
                            <td className="p-5 font-black text-gray-900 uppercase">{r.payer || '---'}</td><td className="p-5 font-black text-teal-700 uppercase">{r.studentName || "---"}</td>
                            <td className="p-5 text-gray-500 italic max-w-[180px] truncate">{r.description || "---"}</td><td className="p-5 text-right font-black text-gray-900 text-base">{r.amount?.toLocaleString() || 0}đ</td>
                            <td className="p-5 text-center"><button onClick={() => setShowInvoice(r)} className="text-orange-600 hover:bg-orange-100 p-2 rounded-xl transition-all"><Printer size={20}/></button></td>
                          </tr>
                        ))
                      )}
                    </tbody>
                  </table>
                </div>
              </div>
            </div>
          );
        })()}
      </main>

      {/* --- CÁC MODALS --- */}

      {/* MODAL 1: HỒ SƠ 360 ĐỘ (CÓ THỐNG KÊ CHUYÊN CẦN) */}
      {showStudentDetail && (() => {
        // Thuật toán tự động đếm số buổi Có mặt / Vắng / Muộn
        const studentAtt = attendanceLogs.filter(a => a["Mã HS"] === showStudentDetail.id);
        const totalSessions = studentAtt.length;
        const presentCount = studentAtt.filter(a => a["Trạng thái"] === "Có mặt").length;
        const lateCount = studentAtt.filter(a => a["Trạng thái"] === "Muộn").length;
        const absentCount = studentAtt.filter(a => a["Trạng thái"] === "Vắng").length;
        const attendedCount = presentCount + lateCount;
        const attRate = totalSessions > 0 ? Math.round((attendedCount / totalSessions) * 100) : 0;

        return (
          <div className="fixed inset-0 z-[100] flex items-center justify-center p-4 bg-black/60 backdrop-blur-md animate-in fade-in">
            <div className="bg-white w-full max-w-5xl rounded-[48px] shadow-2xl overflow-hidden animate-in zoom-in-95 flex flex-col max-h-[90vh]">
              <div className="bg-blue-600 p-8 text-white flex justify-between items-start shrink-0">
                <div className="flex items-center space-x-6">
                  <div className="w-20 h-20 bg-white text-blue-600 rounded-full flex items-center justify-center font-black text-3xl shadow-lg">{showStudentDetail.name ? showStudentDetail.name.charAt(0).toUpperCase() : "?"}</div>
                  <div>
                    <h3 className="text-3xl font-black uppercase leading-none tracking-tight">{showStudentDetail.name || "---"}</h3>
                    <p className="text-blue-100 text-sm mt-2 font-black tracking-widest uppercase"><span className="bg-blue-700 px-3 py-1 rounded-lg mr-2">{showStudentDetail.id || "---"}</span> Lớp: {showStudentDetail.classId || "---"}</p>
                  </div>
                </div>
                <button onClick={() => setShowStudentDetail(null)} className="p-3 bg-white/10 rounded-full hover:bg-white/20 transition-all"><X size={24}/></button>
              </div>
              
              <div className="p-8 overflow-y-auto bg-gray-50 flex-1">
                <div className="grid grid-cols-1 lg:grid-cols-3 gap-8 mb-8">
                  {/* Cột 1: Cá nhân */}
                  <div className="bg-white p-6 rounded-[32px] shadow-sm border border-gray-100 space-y-4">
                    <h4 className="font-black text-gray-400 uppercase tracking-widest text-xs mb-4 flex items-center"><UserCheck size={16} className="mr-2"/> Cá nhân & Học thuật</h4>
                    {[
                      { l: 'Ngày sinh', v: showStudentDetail.dob }, { l: 'Trường học', v: showStudentDetail.school },
                      { l: 'Địa chỉ', v: showStudentDetail.address }, { l: 'Học lực', v: showStudentDetail.academicLevel, c: "text-teal-600 font-black" },
                      { l: 'Mục tiêu', v: showStudentDetail.goal }, { l: 'SĐT Học sinh', v: showStudentDetail.studentPhone },
                      { l: 'Bắt đầu học', v: showStudentDetail.startDate, c: "text-green-600" }, { l: 'Dự kiến kết thúc', v: showStudentDetail.endDate, c: "text-red-600" }
                    ].map((item, idx) => (
                      <div key={idx} className="flex justify-between items-center border-b border-gray-50 pb-2"><span className="text-xs font-bold text-gray-500">{item.l}</span><span className={cn("text-sm font-black text-gray-800 text-right max-w-[150px] truncate", item.c)}>{item.v || '---'}</span></div>
                    ))}
                  </div>
                  
                  {/* Cột 2: Phụ huynh */}
                  <div className="bg-white p-6 rounded-[32px] shadow-sm border border-gray-100 space-y-4">
                    <h4 className="font-black text-gray-400 uppercase tracking-widest text-xs mb-4 flex items-center"><PhoneCall size={16} className="mr-2"/> Phụ huynh & Liên hệ</h4>
                    {[
                      { l: 'Tên phụ huynh', v: showStudentDetail.parentName }, { l: 'SĐT (Zalo)', v: showStudentDetail.parentPhone, c: "text-blue-600 font-black" },
                      { l: 'Cơ sở đào tạo', v: showStudentDetail.branch }, { l: 'Giáo viên', v: showStudentDetail.teacher }
                    ].map((item, idx) => (
                      <div key={idx} className="flex justify-between items-center border-b border-gray-50 pb-2"><span className="text-xs font-bold text-gray-500">{item.l}</span><span className={cn("text-sm font-black text-gray-800 text-right max-w-[150px] truncate", item.c)}>{item.v || '---'}</span></div>
                    ))}
                    <button onClick={() => window.open(`https://zalo.me/${showStudentDetail.parentPhone?.replace(/\D/g,'')}`, '_blank')} className="w-full mt-4 py-3 bg-blue-100 text-blue-700 rounded-2xl font-black uppercase text-xs hover:bg-blue-200 transition-colors flex justify-center items-center"><PhoneCall size={16} className="mr-2"/> Mở Zalo Phụ Huynh</button>
                  </div>

                  {/* Cột 3: THỐNG KÊ CHUYÊN CẦN */}
                  <div className="bg-white p-6 rounded-[32px] shadow-sm border border-gray-100 space-y-4 flex flex-col">
                    <h4 className="font-black text-gray-400 uppercase tracking-widest text-xs mb-4 flex items-center"><PieChartIcon size={16} className="mr-2"/> Thống kê Chuyên cần</h4>
                    <div className="flex-1 flex flex-col justify-center items-center mb-4">
                      <div className={cn("w-32 h-32 rounded-full border-8 flex items-center justify-center mb-4", attRate >= 80 ? "border-green-500 text-green-600" : (attRate >= 50 ? "border-amber-500 text-amber-600" : "border-red-500 text-red-600"))}>
                        <span className="text-3xl font-black">{attRate}%</span>
                      </div>
                      <p className="text-xs font-black text-gray-400 uppercase tracking-widest">TỶ LỆ ĐI HỌC</p>
                    </div>
                    <div className="grid grid-cols-2 gap-2 text-center">
                      <div className="bg-gray-50 p-2 rounded-xl"><p className="text-xs font-bold text-gray-500">Tổng buổi</p><p className="font-black text-gray-900">{totalSessions}</p></div>
                      <div className="bg-green-50 p-2 rounded-xl"><p className="text-xs font-bold text-green-600">Có mặt</p><p className="font-black text-green-700">{presentCount}</p></div>
                      <div className="bg-amber-50 p-2 rounded-xl"><p className="text-xs font-bold text-amber-600">Muộn</p><p className="font-black text-amber-700">{lateCount}</p></div>
                      <div className="bg-red-50 p-2 rounded-xl"><p className="text-xs font-bold text-red-600">Vắng</p><p className="font-black text-red-700">{absentCount}</p></div>
                    </div>
                  </div>
                </div>

                {/* BẢNG LỊCH SỬ THU PHÍ */}
                <div className="bg-white p-6 rounded-[32px] shadow-sm border border-gray-100">
                  <h4 className="font-black text-gray-400 uppercase tracking-widest text-xs mb-4 flex items-center"><History size={16} className="mr-2"/> Lịch sử nộp học phí</h4>
                  <div className="overflow-x-auto">
                    <table className="w-full text-left text-xs">
                      <thead className="text-gray-400 font-bold uppercase border-b"><tr className="pb-2"><th className="pb-2">Ngày CT</th><th className="pb-2">Diễn giải</th><th className="pb-2 text-right">Số tiền</th></tr></thead>
                      <tbody className="divide-y divide-gray-50">
                        {payments.filter(p => p.studentId === showStudentDetail.id).length === 0 ? (
                          <tr><td colSpan={3} className="py-4 text-center text-gray-400 italic">Chưa có dữ liệu đóng học phí.</td></tr>
                        ) : (
                          payments.filter(p => p.studentId === showStudentDetail.id).map((p, i) => (
                            <tr key={i} className="hover:bg-gray-50"><td className="py-3 font-bold">{p.date || "---"}</td><td className="py-3 italic truncate max-w-[300px]">{p.description || "---"}</td><td className="py-3 text-right font-black text-orange-600">+{p.amount?.toLocaleString() || 0}đ</td></tr>
                          ))
                        )}
                      </tbody>
                    </table>
                  </div>
                </div>
              </div>
            </div>
          </div>
        );
      })()}

      {/* MODAL 2: THÊM / SỬA HỌC SINH */}
      {showAddHSModal && (
        <div className="fixed inset-0 z-[100] flex items-center justify-center p-4 bg-black/60 backdrop-blur-md animate-in fade-in">
          <div className="bg-white w-full max-w-4xl rounded-[48px] shadow-2xl p-12 space-y-8 overflow-y-auto max-h-[90vh]">
            <div className="flex justify-between items-center border-b pb-6">
              <h3 className="text-2xl font-black text-blue-600 uppercase italic tracking-tighter">{editingStudent ? "Cập nhật hồ sơ" : "Hồ sơ học sinh NK mới"}</h3>
              <button onClick={() => setShowAddHSModal(false)} className="p-2 hover:bg-gray-100 rounded-full transition-colors"><X className="text-gray-300" /></button>
            </div>
            <div className="grid grid-cols-1 md:grid-cols-3 gap-6 text-left">
              {[
                { n: 'id', l: 'Mã học sinh (*)' }, { n: 'name', l: 'Họ tên học sinh' }, { n: 'dob', l: 'Ngày sinh' },
                { n: 'branch', l: 'Cơ sở' }, { n: 'grade', l: 'Khối lớp' }, { n: 'school', l: 'Trường học' },
                { n: 'teacher', l: 'Giáo viên' }, { n: 'classId', l: 'Mã lớp tham chiếu' }, { n: 'parentName', l: 'Tên phụ huynh' },
                { n: 'parentPhone', l: 'SĐT Phụ huynh' }, { n: 'studentPhone', l: 'SĐT Học sinh' }, { n: 'address', l: 'Địa chỉ' },
                { n: 'academicLevel', l: 'Học lực' }, { n: 'goal', l: 'Mục tiêu' }, { n: 'supportNeeded', l: 'Cần hỗ trợ' },
                { n: 'startDate', l: 'Ngày bắt đầu (DD/MM/YYYY)' }, { n: 'endDate', l: 'Ngày kết thúc (DD/MM/YYYY)' }
              ].map(f => (
                <div key={f.n} className="space-y-1">
                  <p className="text-[10px] font-black text-gray-400 uppercase tracking-widest ml-1">{f.l}</p>
                  <input value={formHS[f.n] || ''} readOnly={editingStudent && f.n === 'id'} className={cn("w-full p-3 bg-gray-50 rounded-xl border-none outline-none focus:ring-2 ring-blue-500 text-sm font-bold text-gray-700", editingStudent && f.n === 'id' && "opacity-50 cursor-not-allowed")} onChange={e => setFormHS({...formHS, [f.n]: e.target.value})} />
                </div>
              ))}
            </div>
            <div className="flex space-x-4 pt-6">
              <button onClick={() => setShowAddHSModal(false)} className="flex-1 py-4 bg-gray-100 rounded-3xl font-black uppercase text-xs tracking-widest text-gray-500 hover:bg-gray-200 transition-all">Hủy bỏ</button>
              <button onClick={handleSaveHS} disabled={isSaving} className={cn("flex-1 py-4 text-white rounded-3xl font-black uppercase text-xs tracking-widest shadow-lg transition-all active:scale-95", isSaving ? "bg-gray-400" : "bg-blue-600 shadow-blue-200 hover:bg-blue-700")}>{isSaving ? "Đang xử lý..." : "Lưu vào hệ thống"}</button>
            </div>
          </div>
        </div>
      )}

      {/* MODAL 3: GHI THU PHÍ (TÌM KIẾM HỌC SINH THÔNG MINH) */}
      {showAddFeeModal && (
        <div className="fixed inset-0 z-[100] flex items-center justify-center p-4 bg-black/60 backdrop-blur-sm animate-in fade-in">
          <div className="bg-white w-full max-w-md rounded-[48px] shadow-2xl p-10 space-y-6 animate-in zoom-in-95">
            <h3 className="text-2xl font-black text-orange-600 uppercase border-b-2 border-orange-100 pb-4 italic tracking-tighter">Ghi sổ thu phí S1a</h3>
            <div className="space-y-4 text-left">
              <div><p className="text-sm font-black text-gray-400 uppercase mb-2 tracking-widest ml-1">Ngày ghi chứng từ</p><input type="date" value={formFee.date} className="w-full p-4 bg-gray-50 rounded-2xl border-none font-black text-lg outline-none shadow-inner" onChange={e => setFormFee({...formFee, date: e.target.value})}/></div>
              <div><p className="text-sm font-black text-gray-400 uppercase mb-2 tracking-widest ml-1">Người nộp</p><input placeholder="Tên phụ huynh nộp phí..." className="w-full p-4 bg-gray-50 rounded-2xl border-none text-base outline-none font-bold shadow-inner" onChange={e => setFormFee({...formFee, payer: e.target.value})}/></div>
              
              {/* Ô TÌM KIẾM HỌC SINH THÔNG MINH (DATALIST) */}
              <div>
                <p className="text-sm font-black text-gray-400 uppercase mb-2 tracking-widest ml-1">Học sinh thụ hưởng</p>
                <input 
                  list="student-search-list" 
                  placeholder="Gõ tên hoặc mã HS..." 
                  className="w-full p-4 bg-gray-50 rounded-2xl border-none font-black text-base outline-none shadow-inner text-teal-700 focus:ring-2 ring-orange-500" 
                  value={formFee._displayHS || ''}
                  onChange={e => {
                    const val = e.target.value;
                    const extractedId = val.includes(' - ') ? val.split(' - ')[0] : val;
                    setFormFee({...formFee, maHS: extractedId, _displayHS: val});
                  }}
                />
                <datalist id="student-search-list">
                  {students.map(s => <option key={s.id} value={`${s.id} - ${s.name}`} />)}
                </datalist>
              </div>

              <div><p className="text-sm font-black text-gray-400 uppercase mb-2 tracking-widest ml-1">Số tiền đóng (VNĐ)</p><input type="number" placeholder="800,000" className="w-full p-4 bg-gray-50 rounded-2xl border-none font-black text-orange-600 text-2xl outline-none shadow-inner focus:ring-2 ring-orange-500" onChange={e => setFormFee({...formFee, soTien: e.target.value})}/></div>
              <div><p className="text-sm font-black text-gray-400 uppercase mb-2 tracking-widest ml-1">Diễn giải</p><input placeholder="Học phí khóa học / Tháng..." className="w-full p-4 bg-gray-50 rounded-2xl border-none text-base outline-none font-bold shadow-inner focus:ring-2 ring-orange-500" onChange={e => setFormFee({...formFee, description: e.target.value})}/></div>
            </div>
            <div className="flex space-x-4 pt-4">
              <button onClick={() => setShowAddFeeModal(false)} className="flex-1 py-5 bg-gray-100 rounded-3xl font-black uppercase text-sm text-gray-500 hover:bg-gray-200 transition-all">Hủy</button>
              <button onClick={handleSaveFee} disabled={isSaving} className={cn("flex-1 py-5 text-white rounded-3xl font-black uppercase text-sm shadow-xl transition-all active:scale-95", isSaving ? "bg-gray-400" : "bg-orange-600 shadow-orange-100 hover:bg-orange-700")}>{isSaving ? "Đang lưu..." : "Xác nhận & Ghi sổ"}</button>
            </div>
          </div>
        </div>
      )}

      {/* MODAL 4: BIÊN LAI XUẤT PDF */}
      {showInvoice && (
        <div className="fixed inset-0 z-[100] flex items-center justify-center p-4 bg-black/60 backdrop-blur-sm animate-in fade-in">
          <div className="bg-white w-full max-w-lg rounded-[40px] shadow-2xl p-10 animate-in zoom-in-95 overflow-hidden relative">
             <div className="absolute top-0 left-0 w-full h-3 bg-orange-600"></div>
             <div className="text-center mb-6"><GraduationCap className="mx-auto text-orange-600 mb-2" size={48}/><h4 className="text-2xl font-black text-gray-900 uppercase tracking-tighter">Lớp Toán NK</h4><p className="text-gray-400 text-xs font-bold uppercase tracking-[0.2em] mt-1">Biên lai thu tiền điện tử</p></div>
             <div className="space-y-3 mb-8 bg-orange-50/50 p-6 rounded-3xl border border-orange-100">
               {[
                 { l: 'Ngày chứng từ', v: showInvoice.date }, { l: 'Số hiệu CT', v: showInvoice.docNum, c: 'text-orange-600 italic' }, { l: 'Mã học sinh', v: showInvoice.studentId },
                 { l: 'Họ và tên', v: showInvoice.studentName.toUpperCase(), c: 'text-lg text-teal-700' }, { l: 'Người thanh toán', v: showInvoice.payer || '---' },
                 { l: 'Hình thức', v: showInvoice.method || '---' }, { l: 'Diễn giải', v: showInvoice.description || '---' },
               ].map((item, idx) => (
                 <div key={idx} className="flex justify-between border-b border-orange-100/50 pb-2"><span className="text-xs font-black text-gray-500 uppercase tracking-widest">{item.l}</span><span className={cn("text-sm font-bold text-gray-900 text-right max-w-[200px]", item.c)}>{item.v || "---"}</span></div>
               ))}
               <div className="flex justify-between items-center pt-2"><span className="text-sm font-black text-gray-900 uppercase tracking-widest">SỐ TIỀN</span><span className="text-3xl font-black text-orange-600 tracking-tighter">{showInvoice.amount?.toLocaleString() || 0}đ</span></div>
             </div>
             <button onClick={() => window.print()} className="w-full py-5 bg-orange-600 text-white rounded-2xl font-black uppercase text-sm shadow-xl mb-4 flex items-center justify-center hover:bg-orange-700 active:scale-95 transition-all"><Printer className="mr-3" size={20}/> In biên lai PDF</button>
             <button onClick={() => setShowInvoice(null)} className="w-full text-center text-gray-400 text-xs font-black uppercase tracking-widest hover:text-gray-600">Đóng lại</button>
          </div>
        </div>
      )}
      
      <footer className="text-center mt-12 pb-12 opacity-30 tracking-[0.4em] font-black uppercase text-[10px] text-gray-500">
        Lớp Toán NK • 2026 • Financial Management System v9.0 Ultimate
      </footer>
    </div>
  );
}
