import React, { useState, useEffect } from 'react';
import { 
  GraduationCap, LayoutDashboard, Users, BookOpen, UserCheck, 
  CreditCard, FileText, Save, Printer, X, Check, 
  Plus, Calendar, TrendingUp, PieChart as PieChartIcon, 
  History, BarChart2, ArrowUpRight
} from 'lucide-react';
import { 
  BarChart, Bar, XAxis, YAxis, CartesianGrid, Tooltip, 
  ResponsiveContainer, PieChart, Pie, Cell, Legend
} from 'recharts';
import { clsx, type ClassValue } from 'clsx';
import { twMerge } from 'tailwind-merge';

// --- TIỆN ÍCH GỘP CLASS TAILWIND ---
function cn(...inputs: ClassValue[]) {
  return twMerge(clsx(inputs));
}

// --- CẤU HÌNH API GOOGLE SCRIPTS ---
const SCRIPT_URL = 'https://script.google.com/macros/s/AKfycbwDDCm7fkcHehvZxMSlqQVlARcJWtQu_qRu33SPWn4XEICAJuOxREqkfo-5kQW-4Flp6w/exec';

// --- ĐỊNH NGHĨA KIỂU DỮ LIỆU (INTERFACE) ---
type Screen = 'overview' | 'teaching' | 'students' | 'fees';

interface Student {
  id: string;               // Mã HS
  name: string;             // Họ và tên học sinh
  dob: string;              // Ngày tháng năm sinh
  branch: string;           // Cơ sở học tập
  grade: string;            // Khối lớp hiện tại
  school: string;           // Trường đang học
  teacher: string;          // Giáo viên trực tiếp giảng dạy
  parentName: string;       // Họ và tên phụ huynh
  parentPhone: string;      // Số điện thoại phụ huynh (Zalo)
  studentPhone: string;     // Số điện thoại học sinh
  address: string;          // Địa chỉ thường trú
  academicLevel: string;    // Học lực môn Toán hiện tại
  goal: string;             // Mục tiêu điểm số học kỳ tới
  supportNeeded: string;    // Kiến thức em cần hỗ trợ thêm
  classId: string;          // Mã Lớp tham chiếu
}

interface S1aRecord {
  id: string;               
  date: string;             // Ngày chứng từ
  docNum: string;           // Số hiệu CT
  description: string;      // Diễn giải
  studentId: string;        // Mã HS
  studentName: string;      
  payer: string;            // Người thanh toán
  method: string;           // Hình thức
  amount: number;           // Số tiền
  note?: string;            
}

// Bảng màu chuẩn thương hiệu Lớp Toán NK
const COLORS = ['#14b8a6', '#3b82f6', '#f59e0b', '#ef4444', '#8b5cf6'];

export default function App() {
  // --- STATES QUẢN LÝ DỮ LIỆU TỔNG ---
  const [activeScreen, setActiveScreen] = useState<Screen>('overview');
  const [loading, setLoading] = useState(true);
  const [students, setStudents] = useState<Student[]>([]);
  const [classes, setClasses] = useState<any[]>([]);
  const [payments, setPayments] = useState<S1aRecord[]>([]);
  const [teachingLogs, setTeachingLogs] = useState<any[]>([]);
  const [financeSummary, setFinanceSummary] = useState<any>(null);

  // --- STATES QUẢN LÝ GIAO DIỆN & MODAL ---
  const [selectedClassId, setSelectedClassId] = useState('');
  const [showInvoice, setShowInvoice] = useState<S1aRecord | null>(null);
  const [showStudentDetail, setShowStudentDetail] = useState<Student | null>(null);
  const [showAddHSModal, setShowAddHSModal] = useState(false);
  const [showAddFeeModal, setShowAddFeeModal] = useState(false);
// --- STATES GIẢNG DẠY (Tách độc lập để không bị mất focus khi gõ) ---
  const [teachingDate, setTeachingDate] = useState(new Date().toISOString().split('T')[0]);
  const [attendance, setAttendance] = useState<Record<string, string>>({});
  const [attendanceNotes, setAttendanceNotes] = useState<Record<string, string>>({});
  const [journalContent, setJournalContent] = useState('');
  const [journalHomework, setJournalHomework] = useState('');

  // --- STATES NHẬP LIỆU BIỂU MẪU ---
  const [formHS, setFormHS] = useState<any>({ branch: 'Đào Tấn', academicLevel: 'Khá' });
  const [formFee, setFormFee] = useState<any>({ method: 'Chuyển khoản', date: new Date().toISOString().split('T')[0] });

  // --- HÀM ĐỊNH DẠNG NGÀY CHUẨN VIỆT NAM (DD/MM/YYYY) ---
  const formatDisplayDate = (dateStr: any) => {
    if (!dateStr) return '---';
    try {
      const d = new Date(dateStr);
      if (isNaN(d.getTime())) return dateStr;
      const day = d.getDate().toString().padStart(2, '0');
      const month = (d.getMonth() + 1).toString().padStart(2, '0');
      const year = d.getFullYear();
      return `${day}/${month}/${year}`;
    } catch (e) { return dateStr; }
  };

  // 1. TẢI DỮ LIỆU TỪ GOOGLE SHEETS (Chống lỗi trắng trang do dữ liệu rỗng)
  const loadData = async () => {
    try {
      setLoading(true);
      const res = await fetch(SCRIPT_URL);
      const data = await res.json();
      
      const mappedHS = (data.students || []).map((s: any) => ({
        id: s["Mã HS"] || "", 
        name: s["Họ và tên học sinh"] || "Chưa cập nhật", 
        dob: formatDisplayDate(s["Ngày tháng năm sinh"]),
        branch: s["Cơ sở học tập"] || "", 
        grade: s["Khối lớp hiện tại"] || "", 
        school: s["Trường đang học"] || "",
        teacher: s["Giáo viên trực tiếp giảng dạy"] || "", 
        parentName: s["Họ và tên phụ huynh"] || "",
        parentPhone: s["Số điện thoại phụ huynh (Zalo)"] || "", 
        studentPhone: s["Số điện thoại học sinh"] || "",
        address: s["Địa chỉ thường trú"] || "", 
        academicLevel: s["Học lực môn Toán hiện tại"] || "",
        goal: s["Mục tiêu điểm số học kỳ tới"] || "", 
        supportNeeded: s["Kiến thức em cần hỗ trợ thêm"] || "",
        classId: s["Mã Lớp"] || ""
      }));

      setStudents(mappedHS);
      setClasses(data.classes || []);
      setFinanceSummary(data.finance || {});
      setTeachingLogs(data.teachingLogs || []);
      
      // Ánh xạ Sổ thu phí & Khớp nối Tên Học Sinh an toàn
      setPayments((data.payments || []).map((p:any, i:number) => {
        const foundStudent = mappedHS.find((s:any) => s.id === p["Mã HS"]);
        const sName = foundStudent ? foundStudent.name : "HS KHÔNG TỒN TẠI";

        return {
          id: i.toString(), 
          date: formatDisplayDate(p["Ngày CT"]), 
          docNum: p["Số hiệu CT"] || "---",
          studentId: p["Mã HS"] || "---", 
          studentName: sName,
          payer: p["Người thanh toán"] || "---",
          method: p["Hình thức"] || "---", 
          description: p["Diễn giải"] || "---",
          amount: Number(p["Số tiền"]) || 0
        };
      }));
      
      if (data.classes?.length > 0 && !selectedClassId) {
        setSelectedClassId(data.classes[0]["Mã Lớp"]);
      }
    } catch (e) { 
      console.error("Lỗi tải dữ liệu:", e);
    } finally { 
      setLoading(false); 
    }
  };

  useEffect(() => { loadData(); }, []);
// 2. LƯU GIẢNG DẠY & LÀM MỚI FORM SAU KHI LƯU
  const handleSaveTeaching = async () => {
    const attRecords = students.filter(s => s.classId === selectedClassId).map(s => ({
      maHS: s.id, tenHS: s.name, trangThai: attendance[s.id] || 'Có mặt', ghiChu: attendanceNotes[s.id] || ''
    }));
    setLoading(true);
    const payload = { 
      action: "saveDaily", date: teachingDate, maLop: selectedClassId, 
      attendance: attRecords, content: journalContent, homework: journalHomework 
    };
    await fetch(SCRIPT_URL, { method: 'POST', body: JSON.stringify(payload) });
    
    // Xóa trắng các trường để sẵn sàng nhập cho lớp khác
    setJournalContent('');
    setJournalHomework('');
    setAttendance({});
    setAttendanceNotes({});
    
    setLoading(false); 
    alert("Đã lưu và làm mới dữ liệu dạy học!"); 
    loadData();
  };

  // 3. LƯU HỌC PHÍ & TẠO MÃ CHỨNG TỪ THEO QUY TẮC
  const handleSaveFee = async () => {
    setLoading(true);
    const now = new Date();
    const hours = now.getHours().toString().padStart(2, '0');
    const minutes = now.getMinutes().toString().padStart(2, '0');
    const saveTime = `${hours}:${minutes} - ${formatDisplayDate(formFee.date)}`;
    
    // Quy tắc mã chứng từ: PT-YYMMDD-MãHS (VD: PT-260305-HS002)
    const yy = now.getFullYear().toString().slice(2);
    const mm = (now.getMonth() + 1).toString().padStart(2, '0');
    const dd = now.getDate().toString().padStart(2, '0');
    const systemDocNum = `PT-${yy}${mm}${dd}-${formFee.maHS}`;

    await fetch(SCRIPT_URL, { method: 'POST', body: JSON.stringify({ 
      action: "savePayment", timeStamp: saveTime, soCT: systemDocNum, ...formFee 
    })});
    setShowAddFeeModal(false); 
    loadData();
  };

  // 4. LƯU HỒ SƠ HỌC SINH MỚI
  const handleSaveHS = async () => {
    setLoading(true);
    await fetch(SCRIPT_URL, { method: 'POST', body: JSON.stringify({ action: "saveHS", ...formHS }) });
    setShowAddHSModal(false); loadData();
  };

  // --- GIAO DIỆN CHỜ (LOADING SCREEN) ---
  if (loading) return (
    <div className="min-h-screen flex flex-col items-center justify-center bg-teal-50 text-teal-600">
      <GraduationCap size={80} className="animate-bounce mb-4" />
      <h1 className="text-xl font-black uppercase italic tracking-widest">Lớp Toán NK đang tải...</h1>
    </div>
  );

  // --- COMPONENT: THANH ĐIỀU HƯỚNG (NAVBAR) ---
  const Navbar = () => (
    <nav className="bg-white border-b border-gray-200 sticky top-0 z-50 shadow-sm">
      <div className="max-w-7xl mx-auto px-4 flex justify-between h-20 items-center">
        <div className="flex items-center text-teal-600">
          <GraduationCap size={40} className="mr-3" />
          <div className="flex flex-col">
            <span className="font-black text-xl uppercase leading-none">Lớp Toán NK</span>
            <span className="text-[10px] font-bold text-gray-400 uppercase tracking-[0.2em] mt-1">Hệ thống quản lý 2026</span>
          </div>
        </div>
        <div className="flex space-x-2">
          {[
            { id: 'overview', label: 'Tổng quan', icon: LayoutDashboard, color: 'text-teal-600' },
            { id: 'teaching', label: 'Giảng dạy', icon: UserCheck, color: 'text-green-600' },
            { id: 'students', label: 'Học sinh', icon: Users, color: 'text-blue-600' },
            { id: 'fees', label: 'Sổ thu phí', icon: CreditCard, color: 'text-orange-500' },
          ].map((item) => (
            <button 
              key={item.id} 
              onClick={() => setActiveScreen(item.id as Screen)}
              className={cn(
                "flex flex-col items-center px-4 py-2 rounded-2xl transition-all", 
                activeScreen === item.id ? "bg-gray-100 " + item.color : "text-gray-400 hover:bg-gray-50"
              )}
            >
              <item.icon size={22} />
              <span className="text-[10px] font-black uppercase mt-1 tracking-wider">{item.label}</span>
            </button>
          ))}
        </div>
      </div>
    </nav>
  );

  return (
    <div className="min-h-screen bg-[#F8FAFC] pb-12 font-sans selection:bg-teal-100">
      <Navbar />
      <main className="max-w-7xl mx-auto px-4 py-8">
{/* --- TAB 1: TỔNG QUAN --- */}
        {activeScreen === 'overview' && (
          <div className="space-y-6 animate-in fade-in duration-700">
            {/* 3 Thẻ số liệu */}
            <div className="grid grid-cols-1 md:grid-cols-3 gap-6">
              <div className="bg-white p-6 rounded-[32px] shadow-sm border border-gray-100 flex items-center space-x-5">
                <div className="p-4 bg-blue-100 text-blue-600 rounded-3xl"><Users size={32}/></div>
                <div><p className="text-sm font-bold text-gray-400 uppercase">Học sinh</p><p className="text-4xl font-black text-gray-900">{students.length}</p></div>
              </div>
              <div className="bg-white p-6 rounded-[32px] shadow-sm border border-gray-100 flex items-center space-x-5">
                <div className="p-4 bg-teal-100 text-teal-600 rounded-3xl"><TrendingUp size={32}/></div>
                <div><p className="text-sm font-bold text-gray-400 uppercase">Doanh thu</p><p className="text-4xl font-black text-gray-900">{financeSummary?.totalRevenue?.toLocaleString()}đ</p></div>
              </div>
              <div className="bg-white p-6 rounded-[32px] shadow-sm border border-gray-100 flex items-center space-x-5">
                <div className="p-4 bg-purple-100 text-purple-600 rounded-3xl"><BookOpen size={32}/></div>
                <div><p className="text-sm font-bold text-gray-400 uppercase">Lớp học</p><p className="text-4xl font-black text-gray-900">{classes.length}</p></div>
              </div>
            </div>

            <div className="grid grid-cols-1 lg:grid-cols-2 gap-6">
              {/* Biểu đồ */}
              <div className="bg-white p-8 rounded-[40px] shadow-sm border border-gray-100">
                <h3 className="font-black uppercase tracking-tight text-gray-800 mb-6 flex items-center"><BarChart2 size={20} className="mr-2 text-teal-600"/> Doanh thu Giáo viên</h3>
                <div className="h-[220px]"><ResponsiveContainer><BarChart data={[{name:'Thầy Nhân', r: financeSummary?.revenueByTeacher?.nhan || 0}, {name:'Cô Kiên', r: financeSummary?.revenueByTeacher?.kien || 0}]}><CartesianGrid strokeDasharray="3 3" vertical={false}/><XAxis dataKey="name"/><YAxis/><Tooltip cursor={{fill:'#f0fdfa'}}/><Bar dataKey="r" fill="#14b8a6" radius={[10,10,0,0]} barSize={50}/></BarChart></ResponsiveContainer></div>
              </div>
              <div className="bg-white p-8 rounded-[40px] shadow-sm border border-gray-100"><h3 className="font-black uppercase tracking-tight text-gray-800 mb-6 flex items-center"><PieChartIcon size={20} className="mr-2 text-orange-600"/> Cơ cấu Học lực</h3>
                <div className="h-[220px]"><ResponsiveContainer><PieChart><Pie data={[{name:'Giỏi',v:financeSummary?.gradeStats?.Gioi},{name:'Khá',v:financeSummary?.gradeStats?.Kha},{name:'TB',v:financeSummary?.gradeStats?.TrungBinh}]} cx="50%" cy="50%" innerRadius={60} outerRadius={90} paddingAngle={8} dataKey="v">{COLORS.map((c,i)=><Cell key={i} fill={c}/>)}</Pie><Tooltip/><Legend verticalAlign="bottom"/></PieChart></ResponsiveContainer></div>
              </div>

              {/* BẢNG 1: HOẠT ĐỘNG THU PHÍ (Đã tách riêng) */}
              <div className="bg-white rounded-[40px] border border-orange-100 overflow-hidden shadow-sm">
                <div className="bg-orange-50 p-5 border-b border-orange-100"><h3 className="font-black uppercase text-orange-700 flex items-center"><CreditCard size={18} className="mr-2"/> Thu phí gần đây</h3></div>
                <table className="w-full text-left text-sm"><tbody className="divide-y divide-gray-50">
                  {payments.slice(0, 5).map((p, i) => (
                    <tr key={i} className="hover:bg-orange-50/20"><td className="p-4 pl-6 font-bold text-gray-500 whitespace-nowrap">{p.date}</td><td className="p-4 font-black uppercase text-gray-900">{p.studentName}</td><td className="p-4 pr-6 text-right font-black text-orange-600">+{p.amount.toLocaleString()}đ</td></tr>
                  ))}
                </tbody></table>
              </div>

              {/* BẢNG 2: HOẠT ĐỘNG GIẢNG DẠY (Đã tách riêng) */}
              <div className="bg-white rounded-[40px] border border-green-100 overflow-hidden shadow-sm">
                <div className="bg-green-50 p-5 border-b border-green-100"><h3 className="font-black uppercase text-green-700 flex items-center"><UserCheck size={18} className="mr-2"/> Dạy học gần đây</h3></div>
                <table className="w-full text-left text-sm"><tbody className="divide-y divide-gray-50">
                  {teachingLogs.slice(0, 5).map((t, i) => (
                    <tr key={i} className="hover:bg-green-50/20"><td className="p-4 pl-6 font-bold text-gray-500 whitespace-nowrap">{formatDisplayDate(t["Ngày"])}</td><td className="p-4 font-black uppercase text-teal-700">{t["Mã Lớp"]}</td><td className="p-4 pr-6 text-gray-600 italic truncate max-w-[150px]">{t["Nội dung bài dạy"] || '---'}</td></tr>
                  ))}
                </tbody></table>
              </div>
            </div>
          </div>
        )}

        {/* --- TAB 2: GIẢNG DẠY (CỠ CHỮ TO - ĐÃ SỬA ID TIẾNG VIỆT) --- */}
        {activeScreen === 'teaching' && (
          <div className="space-y-6 animate-in slide-in-from-bottom-8">
            <div className="bg-green-600 p-8 rounded-[40px] text-white shadow-xl flex flex-col md:flex-row justify-between items-center gap-4 border-b-8 border-green-700">
              <div><h2 className="text-3xl font-black uppercase italic tracking-tighter">Sổ giảng dạy trực tuyến</h2><p className="text-green-100 text-base font-bold italic">Mặc định: Có mặt</p></div>
              <div className="flex items-center space-x-3 w-full md:w-auto">
                <div className="bg-white/20 px-5 py-3 rounded-2xl flex items-center shadow-inner"><Calendar size={20} className="mr-2" /><input type="date" value={teachingDate} onChange={e=>setTeachingDate(e.target.value)} className="bg-transparent text-white outline-none font-black text-lg [color-scheme:dark]" /></div>
                <select value={selectedClassId} onChange={e=>setSelectedClassId(e.target.value)} className="bg-white text-green-700 px-6 py-4 rounded-2xl font-black text-lg outline-none shadow-2xl">
                  {classes.map(c => <option key={c["Mã Lớp"]} value={c["Mã Lớp"]}>Mã Lớp: {c["Mã Lớp"]}</option>)}
                </select>
              </div>
            </div>

            <div className="grid grid-cols-1 lg:grid-cols-3 gap-6">
              <div className="lg:col-span-2 bg-white rounded-[40px] border border-green-100 overflow-hidden shadow-xl">
                <table className="w-full text-left">
                  <thead className="bg-green-50 text-green-700 font-black uppercase tracking-widest border-b border-green-100">
                    <tr className="text-lg">
                      <th className="p-6">THÔNG TIN HỌC SINH</th>
                      <th className="p-6 text-center">TRẠNG THÁI</th>
                      <th className="p-6">GHI CHÚ</th>
                    </tr>
                  </thead>
                  <tbody className="divide-y divide-gray-100">
                    {students.filter(s => s.classId === selectedClassId).map(s => (
                      <tr key={s.id} className="hover:bg-green-50/30 transition-all">
                        <td className="p-6">
                           <p className="font-black text-gray-900 uppercase text-lg leading-none mb-1">{s.name}</p>
                           <p className="text-lg text-teal-600 font-black tracking-widest">{s.id}</p>
                        </td>
                        <td className="p-6">
                          <div className="flex justify-center space-x-2">
                            {/* ĐÃ SỬA ID THÀNH TIẾNG VIỆT CHUẨN */}
                            {[
                              {id:'Có mặt',l:'Có mặt',c:'bg-green-500'},
                              {id:'Muộn',l:'Muộn',c:'bg-amber-500'},
                              {id:'Vắng',l:'Vắng',c:'bg-red-500'}
                            ].map(opt=>(
                              <button key={opt.id} onClick={()=>setAttendance(p=>({...p, [s.id]: opt.id}))} 
                                className={cn(
                                  "px-5 py-3 rounded-2xl text-sm font-black uppercase transition-all shadow-sm active:scale-90", 
                                  (attendance[s.id]===opt.id || (!attendance[s.id] && opt.id==='Có mặt')) 
                                    ? opt.c+" text-white shadow-lg ring-4 ring-white" 
                                    : "bg-gray-100 text-gray-400 hover:bg-gray-200"
                                )}>
                                {opt.l}
                              </button>
                            ))}
                          </div>
                        </td>
                        <td className="p-6">
                          <input placeholder="Ghi chú buổi học..." className="w-full bg-gray-50 p-4 rounded-xl text-base font-bold outline-none focus:ring-2 ring-green-500 shadow-inner" value={attendanceNotes[s.id] || ''} onChange={e=>setAttendanceNotes({...attendanceNotes, [s.id]: e.target.value})}/>
                        </td>
                      </tr>
                    ))}
                  </tbody>
                </table>
                <div className="p-10 bg-gray-50 border-t border-gray-100 flex justify-end">
                  <button onClick={handleSaveTeaching} className="bg-green-600 text-white px-12 py-5 rounded-[24px] font-black uppercase text-lg tracking-widest shadow-2xl hover:bg-green-700 active:scale-95 transition-all flex items-center">
                    <Save size={24} className="mr-3"/> Lưu toàn bộ dữ liệu dạy học
                  </button>
                </div>
              </div>

              <div className="bg-white p-8 rounded-[40px] border border-green-100 space-y-8 shadow-xl">
                <h3 className="font-black text-green-700 text-xl uppercase flex items-center border-b-2 border-green-100 pb-5">
                  <FileText size={24} className="mr-3"/> Nhật ký buổi dạy
                </h3>
                <div className="space-y-6">
                  <div>
                    <p className="text-sm font-black text-gray-400 uppercase tracking-widest mb-3">Nội dung giảng dạy</p>
                    <textarea placeholder="Hôm nay các em học gì?" className="w-full p-5 bg-gray-50 rounded-[32px] h-48 text-lg font-bold outline-none border-none focus:ring-2 ring-green-500 shadow-inner leading-relaxed" value={journalContent} onChange={e=>setJournalContent(e.target.value)}/>
                  </div>
                  <div>
                    <p className="text-sm font-black text-gray-400 uppercase tracking-widest mb-3">Bài tập về nhà</p>
                    <input placeholder="Giao bài tập về nhà..." className="w-full p-5 bg-gray-50 rounded-2xl text-lg font-bold border-none focus:ring-2 ring-green-500 shadow-inner" value={journalHomework} onChange={e=>setJournalHomework(e.target.value)}/>
                  </div>
                </div>
              </div>
            </div>
          </div>
        )}
{/* --- TAB 3: HỌC SINH (DẠNG BẢNG NGANG - CỠ CHỮ TO) --- */}
        {activeScreen === 'students' && (
